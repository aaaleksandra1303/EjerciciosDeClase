Class Agenda{
	Integer contactos []
	Integer cantidad
	Integer max

	Agenda(){
		cantidad = 0
		max = 20
	}

	Agenda(Integer n){
		cantidad = 0 
		max = n
	}

	AnadirContacto(String nombre){
		if (QuedaEspacio()){
			contactos[cantidad] = nombre
			cantidad++
		}
		else{
			Print "Se ha llegado al límite de contactos."
		}

	}

	EliminarContacto(Integer pos){
		if (pos < cantidad && pos >= 0){
			for (Integer i = pos ; i < cantidad ; i++){
				contactos [i] = contactos[i + 1]
			}
			cantidad--
		}
		else{
			Print "No ha introducido una posición válida o no hay ningún contacto en esa posición"
		}
	}

	BuscarContacto(Integer pos){
		if (pos < cantidad && pos >=0){
			Print "El contacto situado en la posición ", pos, "es", contactos[pos]
		}
		else{
			Print "No ha introducido una posición válida o no hay ningún contacto en esa posición"
		}
	}

	//Devuelve un valor booleano
	QuedaEspacio(){
		if (cantidad >= max){
			return False
		}
		else{
			return True
		} 
	}
}

Inicio(){

	Boolean bucle_salir = True

		}
	}
}
